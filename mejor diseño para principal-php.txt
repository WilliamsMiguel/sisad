            // Función para cargar contenido via AJAX con jQuery
            function cargarContenido(menu_id) {
                // Mostrar indicador de carga
                $('#contenido_dinamico').html(`
            <div class="d-flex justify-content-center align-items-center" style="height: 100%;">
                <div class="loading-spinner"></div>
            </div>
        `);

                $.ajax({
                    url: 'contenido_menu.php',
                    type: 'POST',
                    data: { id_menu: menu_id },
                    success: function(response) {
                        setTimeout(() => {
                            $('#contenido_dinamico').html(response);
                        }, 500);
                    },
                    error: function() {
                        $('#contenido_dinamico').html(`
                    <div class="d-flex justify-content-center align-items-center" style="height: 100%;">
                        <div class="alert alert-danger">
                            <i class="fas fa-exclamation-triangle me-2"></i>
                            Error al cargar el contenido
                        </div>
                    </div>
                `);
                    }
                });
            }

                });

                // Efectos de hover mejorados

                    function() {
                        $(this).find('i').addClass('fa-bounce');
                    },
                    function() {
                        $(this).find('i').removeClass('fa-bounce');
                    }
                );
            });

            // Función para crear partículas animadas
            function createParticles() {
                const particlesContainer = $('<div class="particles"></div>');
                $('body').append(particlesContainer);

                for (let i = 0; i < 50; i++) {
                    const particle = $('<div class="particle"></div>');
                    particle.css({
                        left: Math.random() * 100 + '%',
                        animationDelay: Math.random() * 15 + 's',
                        animationDuration: (Math.random() * 10 + 10) + 's'
                    });
                    particlesContainer.append(particle);
                }
            }
